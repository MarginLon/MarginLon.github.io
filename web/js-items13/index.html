const check = value => {
    value = +value;
    return isNaN(value) ? 0 : value;
};
Number.prototype.plus = function(value){
    // this->n [对象数据类型]
    value = check(value);
    return this + value;
}
Number.prototype.minus = function(value){
    // this->n [对象数据类型]
    value = check(value);
    return this - value;
}
